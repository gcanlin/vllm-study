3. initialize_kv_cache — 分页 KV 池的几何

def initialize_kv_cache(self, kv_cache_config, is_profiling=False) -> None:
    kv_cache_config = deepcopy(kv_cache_config)
    self.kv_cache_config = kv_cache_config

    self.may_add_encoder_only_layers_to_kv_cache_config()
    self.maybe_add_kv_sharing_layers_to_kv_cache_groups(kv_cache_config)

    # 1) 给每个 attention 层选 backend（FlashAttn / FlashInfer / Triton…）
    self.initialize_attn_backend(kv_cache_config, is_profiling=is_profiling)

    # 2) 决定每个 group 的 "kernel block size"（kv_manager 是 256 但 kernel 只支持 64,
    #    则要拆 1→4）
    kernel_block_sizes = prepare_kernel_block_sizes(kv_cache_config, self.attn_groups)

    # 3) 给每个 attention group 建 metadata builder
    self.initialize_metadata_builders(kv_cache_config, kernel_block_sizes)

    # 4) 如果 block_size 变了，重建 InputBatch（block_table 跟着改）
    self.may_reinitialize_input_batch(kv_cache_config, kernel_block_sizes)

    # 5) 真正分配显存
    kv_caches = self.initialize_kv_cache_tensors(kv_cache_config, kernel_block_sizes)

5. execute_model 全景

核心代码骨架（gpu_model_runner.py:3955–4317）：

def execute_model(self, scheduler_output, intermediate_tensors=None):
    if self.execute_model_state is not None:
        raise RuntimeError("State error: sample_tokens() must be called after execute_model().")

    # 这里省略 ngram_gpu / kv_transfer 处理 ...

    num_scheduled_tokens = scheduler_output.total_num_scheduled_tokens
    with record_function_or_nullcontext("gpu_model_runner: preprocess"), \
         self.synchronize_input_prep():

        # ① 同步 InputBatch
        deferred_state_corrections_fn = self._update_states(scheduler_output)

        if not num_scheduled_tokens:
            return EMPTY_MODEL_RUNNER_OUTPUT      # 没活干，直接返回

        num_reqs = self.input_batch.num_reqs
        req_ids = self.input_batch.req_ids
        tokens = [scheduler_output.num_scheduled_tokens[i] for i in req_ids]
        num_scheduled_tokens_np = np.array(tokens, dtype=np.int32)
        max_num_scheduled_tokens = int(num_scheduled_tokens_np.max())
        num_tokens_unpadded = scheduler_output.total_num_scheduled_tokens

        # ② 构建扁平张量
        logits_indices, spec_decode_metadata = self._prepare_inputs(
            scheduler_output, num_scheduled_tokens_np,
        )

        # ③ CUDA graph mode / DBO 分支 / padding
        (cudagraph_mode, batch_desc, should_ubatch,
         num_tokens_across_dp, cudagraph_stats) = self._determine_batch_execution_and_padding(...)

        # ④ 取出每个 KV-group 的 slot_mapping
        slot_mappings_by_group, slot_mappings = self._get_slot_mappings(...)

        # ⑤ 每层 attention 的元数据
        attn_metadata, spec_decode_common_attn_metadata = self._build_attention_metadata(
            num_tokens=num_tokens_unpadded, num_reqs=num_reqs,
            max_query_len=max_num_scheduled_tokens,
            slot_mappings=slot_mappings_by_group, ...
        )

        # ⑥ 把 input_ids / positions / intermediate_tensors 整理好
        (input_ids, inputs_embeds, positions,
         intermediate_tensors, model_kwargs, ec_connector_output) = self._preprocess(
            scheduler_output, num_tokens_padded, intermediate_tensors,
        )

    # ⑦ 进入 forward
    with set_forward_context(
            attn_metadata, self.vllm_config,
            num_tokens=num_tokens_padded,
            num_tokens_across_dp=num_tokens_across_dp,
            cudagraph_runtime_mode=cudagraph_mode,
            batch_descriptor=batch_desc,
            ubatch_slices=ubatch_slices_padded,
            slot_mapping=slot_mappings,
            skip_compiled=has_encoder_input,
         ), \
         record_function_or_nullcontext("gpu_model_runner: forward"), \
         self.maybe_get_kv_connector_output(...) as kv_connector_output:

        model_output = self._model_forward(
            input_ids=input_ids,
            positions=positions,
            intermediate_tensors=intermediate_tensors,
            inputs_embeds=inputs_embeds,
            **model_kwargs,
        )

    # ⑧ Postprocess
    with record_function_or_nullcontext("gpu_model_runner: postprocess"):
        hidden_states = model_output
        if not get_pp_group().is_last_rank:
            return hidden_states                  # PP 中间 rank 不算 logits
        sample_hidden_states = hidden_states[logits_indices]
        logits = self.model.compute_logits(sample_hidden_states)

    # ⑨ 把所有"中间产物"存进 self.execute_model_state，等 sample_tokens() 用
    self.execute_model_state = ExecuteModelState(
        scheduler_output, logits, spec_decode_metadata,
        spec_decode_common_attn_metadata, hidden_states,
        sample_hidden_states, aux_hidden_states,
        ec_connector_output, cudagraph_stats, slot_mappings,
    )
    self.kv_connector_output = kv_connector_output
    return None      # 必须接着调 sample_tokens()

@contextmanager
def override_forward_context(forward_context):
    global _forward_context
    prev_context = _forward_context
    _forward_context = forward_context
    try:
        yield
    finally:
        _forward_context = prev_context

"全局变量"听起来 hack，但因为：

11. 一次完整 step 的时序

12. 总结：12 句话记住整个 model runner

  1. vLLM 把"调度"和"跑模型"分开：Scheduler 决策、ModelRunner 执行。
  2. 启动一次性：load_model（参数进 GPU）→ profile_run（量峰值）→ initialize_kv_cache（建分页池）→ capture_model（CUDA Graph）。
  3. 持久态：self.requests 存所有活跃 req 的完整状态；self.input_batch 存"当前要跑"的 req 的紧凑视图。
  4. 所有 GPU 张量都是预分配的、每步在前缀写新数据——为了 CUDA Graph 能复用地址。
  5. _update_states 把 SchedulerOutput 翻译成 input_batch 的增删改；不动 GPU 张量。
  6. _prepare_inputs 是核心：用 numpy 算 req_indices / positions / token_indices，torch.index_select 把 token id 抽出来，再 H2D。
  7. 关键扁平张量：input_ids[N]、positions[N]、query_start_loc[R+1]、seq_lens[R]、slot_mapping[N]、block_table[R, max_blocks]、logits_indices[R]。
  8. slot_mapping 是写地图（K/V 写到 KV 池哪里），block_table 是读地图（attention 读哪些 block）。
  9. _build_attention_metadata 把上面这堆 wrap 成 dict[layer_name → AttentionMetadata]。
  10. set_forward_context 把 metadata 写进全局变量；模型 forward 不变签名，attention 层在内部用 get_forward_context() + layer_name 取自己那份。
  11. forward 完后 hidden_states[logits_indices] 抽出"要采样的位置" → compute_logits → sampler。
  12. _bookkeeping_sync 把 sampled token 写回 input_batch，把 num_computed_tokens 推进；返回 ModelRunnerOutput 给 Scheduler，开始下一轮。

理解了这 12 点，再去看 multimodal encoder、speculative decoding、PP 跨 rank 通信、async scheduling、KV connector，本质都是在上面这条流水线的某个环节"插入额外步骤"，主线骨架不变。
